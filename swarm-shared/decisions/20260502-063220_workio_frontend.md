import axios from 'axios';

const hfCdnBypass = async (repo, path) => {
  const url = `https://huggingface.co/datasets/${repo}/resolve/main/${path}`;
  const response = await axios.get(url, { responseType: 'arraybuffer' });

## HF CDN Bypass

### Description

The HF CDN bypass allows for downloading public dataset files from the Hugging Face CDN without authentication.

### Usage

```javascript
import hfCdnBypass from '../utils/hfCdnBypass';

const repo = 'your-repo-name';
const path = 'your-path';
const data = await hfCdnBypass(repo, path);
```
### Parameters

* `repo`: the name of the Hugging Face repository
* `path`: the path to the dataset file

### Returns

* the downloaded dataset file as a Buffer
