```python
import requests
import json

def get_cdn_file_path(repo, path):
    return f"https://huggingface.co/datasets/{repo}/resolve/main/{path}"

def download_cdn_file(repo, path):
    file_path = get_cdn_file_path(repo, path)
    response = requests.get(file_path)
    return response.content

def list_repo_tree(repo, path, recursive=False):
    file_list = []
    response = requests.get(f"https://huggingface.co/api/datasets/{repo}/resolve/main/{path}")
    if response.status_code == 200:

    return file_list
```

**API Documentation:**

`cdn_bypass.py` file contains the following API endpoints:

* `get_cdn_file_path(repo, path)`: Returns the CDN file path for a given repository and path.
